#!/usr/bin/env bash

get_config() {
  local file="$1" key="$2"
  grep "^${key}=" "$file" | cut -d'=' -f2-
}

extract_value() {
  local input="$1" key="$2"
  printf '%s\n' "$input" | grep "^${key}=" | cut -d'=' -f2-
}

failures=0
tests=0

assert_eq() {
  local description="$1" expected="$2" actual="$3"
  tests=$((tests + 1))
  if [ "$expected" = "$actual" ]; then
    printf 'PASS: %s\n' "$description"
  else
    printf 'FAIL: %s\n  expected: [%s]\n  actual:   [%s]\n' "$description" "$expected" "$actual"
    failures=$((failures + 1))
  fi
}

# ---------------------------------------------------------------------------
# PURE TESTS (no filesystem, no IO — test the core parsing logic directly)
# These exercise the same grep+cut logic using stdin instead of a file.
# Purity: pure | Extent: core parsing logic
# ---------------------------------------------------------------------------

result=$(extract_value "host=localhost" "host")
assert_eq "pure: simple key=value" "localhost" "$result"

result=$(extract_value "port=8080" "port")
assert_eq "pure: numeric value" "8080" "$result"

result=$(extract_value "db=host=primary=1" "db")
assert_eq "pure: value containing equals signs" "host=primary=1" "$result"

result=$(extract_value "name=" "name")
assert_eq "pure: empty value" "" "$result"

result=$(extract_value "greeting=hello world" "greeting")
assert_eq "pure: value with spaces" "hello world" "$result"

config="alpha=1
beta=2
gamma=3"
result=$(extract_value "$config" "beta")
assert_eq "pure: key in middle of multiline" "2" "$result"

result=$(extract_value "$config" "missing")
assert_eq "pure: missing key returns empty" "" "$result"

result=$(extract_value "# host=commented" "# host")
assert_eq "pure: commented line not matched by bare key" "" "$(extract_value "# host=commented" "host")"

result=$(extract_value "hostname=server1" "host")
assert_eq "pure: partial key match rejected (host vs hostname)" "" "$result"

config="dup=first
dup=second"
result=$(extract_value "$config" "dup")
assert_eq "pure: duplicate keys returns both lines" "first
second" "$result"

result=$(extract_value "path=/usr/local/bin" "path")
assert_eq "pure: value with slashes" "/usr/local/bin" "$result"

result=$(extract_value 'url=https://example.com?q=1&r=2' "url")
assert_eq "pure: value with URL characters" 'https://example.com?q=1&r=2' "$result"

result=$(extract_value "key=  leading spaces" "key")
assert_eq "pure: value with leading spaces preserved" "  leading spaces" "$result"

result=$(extract_value "" "anything")
assert_eq "pure: empty input returns empty" "" "$result"

# ---------------------------------------------------------------------------
# FILESYSTEM TESTS (minimal impurity — exercises get_config with a real file)
# Purity: filesystem | Extent: full function including file read
# ---------------------------------------------------------------------------

tmpdir=$(mktemp -d)
# Clean up temp files on exit to avoid polluting the filesystem
trap 'rm -rf "$tmpdir"' EXIT

config_file="$tmpdir/test.conf"
cat > "$config_file" << 'EOF'
host=localhost
port=8080
db_name=mydb
connection=host=db.local=primary
empty_val=
path=/usr/local/bin
greeting=hello world
EOF

result=$(get_config "$config_file" "host")
assert_eq "file: read host" "localhost" "$result"

result=$(get_config "$config_file" "port")
assert_eq "file: read port" "8080" "$result"

result=$(get_config "$config_file" "connection")
assert_eq "file: value with equals" "host=db.local=primary" "$result"

result=$(get_config "$config_file" "empty_val")
assert_eq "file: empty value" "" "$result"

result=$(get_config "$config_file" "nonexistent")
assert_eq "file: missing key" "" "$result"

result=$(get_config "$config_file" "path")
assert_eq "file: value with slashes" "/usr/local/bin" "$result"

result=$(get_config "$config_file" "ho")
assert_eq "file: partial key not matched" "" "$result"

# Empty file
empty_file="$tmpdir/empty.conf"
touch "$empty_file"
result=$(get_config "$empty_file" "key")
assert_eq "file: empty file returns empty" "" "$result"

# ---------------------------------------------------------------------------
# Summary
# ---------------------------------------------------------------------------

printf '\n%d tests run, %d passed, %d failed\n' "$tests" "$((tests - failures))" "$failures"
exit "$failures"
