cat > test_get_config.sh << 'SCRIPT'
#!/usr/bin/env bash
set -euo pipefail

# Function under test
get_config() {
  local file="$1" key="$2"
  grep "^${key}=" "$file" | cut -d'=' -f2-
}

failures=0
test_num=0

assert_eq() {
  local description="$1" expected="$2" actual="$3"
  test_num=$((test_num + 1))
  if [[ "$expected" == "$actual" ]]; then
    echo "ok $test_num - $description"
  else
    echo "not ok $test_num - $description"
    echo "#   expected: '$expected'"
    echo "#   actual:   '$actual'"
    failures=$((failures + 1))
  fi
}

# Create a temporary directory for config files, clean up on exit
tmpdir=$(mktemp -d)
trap 'rm -rf "$tmpdir"' EXIT

# --- Test: simple key-value retrieval ---
cfg="$tmpdir/simple.cfg"
printf 'name=alice\nage=30\ncity=berlin\n' > "$cfg"

result=$(get_config "$cfg" "name")
assert_eq "simple key retrieval (name)" "alice" "$result"

result=$(get_config "$cfg" "age")
assert_eq "simple key retrieval (age)" "30" "$result"

result=$(get_config "$cfg" "city")
assert_eq "simple key retrieval (city)" "berlin" "$result"

# --- Test: missing key returns empty string ---
result=$(get_config "$cfg" "missing")
assert_eq "missing key returns empty" "" "$result"

# --- Test: value containing equals sign ---
cfg2="$tmpdir/equals.cfg"
printf 'formula=a=b+c\n' > "$cfg2"

result=$(get_config "$cfg2" "formula")
assert_eq "value containing equals sign" "a=b+c" "$result"

# --- Test: empty value ---
cfg3="$tmpdir/empty_val.cfg"
printf 'empty=\nother=stuff\n' > "$cfg3"

result=$(get_config "$cfg3" "empty")
assert_eq "empty value" "" "$result"

# --- Test: value with spaces ---
cfg4="$tmpdir/spaces.cfg"
printf 'greeting=hello world\n' > "$cfg4"

result=$(get_config "$cfg4" "greeting")
assert_eq "value with spaces" "hello world" "$result"

# --- Test: key that is a prefix of another key ---
cfg5="$tmpdir/prefix.cfg"
printf 'host=alpha\nhostname=beta\nhost_port=9999\n' > "$cfg5"

result=$(get_config "$cfg5" "host")
assert_eq "key prefix - exact match only for 'host'" "alpha" "$result"

result=$(get_config "$cfg5" "hostname")
assert_eq "key prefix - 'hostname' is distinct" "beta" "$result"

# --- Test: duplicate keys returns all values (last wins behavior check) ---
cfg6="$tmpdir/dupes.cfg"
printf 'key=first\nkey=second\n' > "$cfg6"

result=$(get_config "$cfg6" "key")
expected=$(printf 'first\nsecond')
assert_eq "duplicate keys returns all matching lines" "$expected" "$result"

# --- Test: value with special characters ---
cfg7="$tmpdir/special.cfg"
printf 'path=/usr/local/bin:/usr/bin\nurl=https://example.com?q=1&r=2\n' > "$cfg7"

result=$(get_config "$cfg7" "path")
assert_eq "value with colons" "/usr/local/bin:/usr/bin" "$result"

result=$(get_config "$cfg7" "url")
assert_eq "value with url characters" "https://example.com?q=1&r=2" "$result"

# --- Test: single entry file ---
cfg8="$tmpdir/single.cfg"
printf 'only=one\n' > "$cfg8"

result=$(get_config "$cfg8" "only")
assert_eq "single entry file" "one" "$result"

# --- Test: empty file ---
cfg9="$tmpdir/empty.cfg"
: > "$cfg9"

result=$(get_config "$cfg9" "anything")
assert_eq "empty file returns empty" "" "$result"

# --- Test: key with no trailing newline ---
cfg10="$tmpdir/nonewline.cfg"
printf 'last=value' > "$cfg10"

result=$(get_config "$cfg10" "last")
assert_eq "file without trailing newline" "value" "$result"

# --- Test: lines that don't match key=value format are ignored ---
cfg11="$tmpdir/mixed.cfg"
printf '# comment\n\nvalid=yes\nnot a pair\n' > "$cfg11"

result=$(get_config "$cfg11" "valid")
assert_eq "ignores non-matching lines" "yes" "$result"

# --- Summary ---
echo ""
echo "Tests run: $test_num, Failures: $failures"

if [[ $failures -gt 0 ]]; then
  exit 1
fi
exit 0
SCRIPT
